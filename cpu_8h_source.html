36 // Helper function to count set bits in the processor mask
37  DWORD CountSetBits(ULONG_PTR pBitMask)
38  {
39  DWORD dwLeftShift = sizeof(ULONG_PTR) * 8 - 1;
40  DWORD dwBitSetCount = 0;
41  ULONG_PTR pBitTest = (ULONG_PTR)1 << dwLeftShift;
42 
43  for (DWORD i = 0; i <= dwLeftShift; ++i)
44  {
45  dwBitSetCount += ((pBitMask & pBitTest) ? 1 : 0);
46  pBitTest /= 2;
47  }
48 
49  return dwBitSetCount;
50  }

154  static std::pair<int, int> TotalCores() {
155 #if defined(__APPLE__)
156  int logical = 0;
157  size_t logical_size = sizeof(logical);
158  if (sysctlbyname("hw.logicalcpu", &logical, &logical_size, nullptr, 0) != 0)
159  logical = -1;
160 
161  int physical = 0;
162  size_t physical_size = sizeof(physical);
163  if (sysctlbyname("hw.physicalcpu", &physical, &physical_size, nullptr, 0) != 0)
164  physical = -1;
165 
166  return std::make_pair(logical, physical);
167 #elif defined(unix) || defined(__unix) || defined(__unix__)
168  long processors = sysconf(_SC_NPROCESSORS_ONLN);
169  return std::make_pair(processors, processors);
170 #elif defined(_WIN32) || defined(_WIN64)
171  BOOL allocated = FALSE;
172  PSYSTEM_LOGICAL_PROCESSOR_INFORMATION pBuffer = nullptr;
173  DWORD dwLength = 0;
174 
175  while (!allocated)
176  {
177  BOOL bResult = GetLogicalProcessorInformation(pBuffer, &dwLength);
178  if (bResult == FALSE)
179  {
180  if (GetLastError() == ERROR_INSUFFICIENT_BUFFER)
181  {
182  if (pBuffer != nullptr)
183  std::free(pBuffer);
184  pBuffer = (PSYSTEM_LOGICAL_PROCESSOR_INFORMATION)std::malloc(dwLength);
185  if (pBuffer == nullptr)
186  return std::make_pair(-1, -1);
187  }
188  else
189  return std::make_pair(-1, -1);
190  }
191  else
192  allocated = TRUE;
193  }
194 
195  std::pair<int, int> result(0, 0);
196  PSYSTEM_LOGICAL_PROCESSOR_INFORMATION pCurrent = pBuffer;
197  DWORD dwOffset = 0;
198 
199  while (dwOffset + sizeof(SYSTEM_LOGICAL_PROCESSOR_INFORMATION) <= dwLength)
200  {
201  switch (pCurrent->Relationship)
202  {
203  case RelationProcessorCore:
204  result.first += Internals::CountSetBits(pCurrent->ProcessorMask);
205  result.second += 1;
206  break;
207  case RelationNumaNode:
208  case RelationCache:
209  case RelationProcessorPackage:
210  break;
211  default:
212  return std::make_pair(-1, -1);
213  }
214  dwOffset += sizeof(SYSTEM_LOGICAL_PROCESSOR_INFORMATION);
215  pCurrent++;
216  }
217 
218  std::free(pBuffer);
219 
220  return result;
221 #else
222 #error Unsupported platform
223 #endif
224  }
225  static int64_t ClockSpeed() {
226 #if defined(__APPLE__)
227  uint64_t frequency = 0;
228  size_t size = sizeof(frequency);
229  if (sysctlbyname("hw.cpufrequency", &frequency, &size, nullptr, 0) == 0)
230  return frequency;
231 
232  return -1;
233 #elif defined(unix) || defined(__unix) || defined(__unix__)
234  static std::regex pattern("cpu MHz(.*): (.*)");
235 
236  std::string line;
237  std::ifstream stream("/proc/cpuinfo");
238  while (getline(stream, line))
239  {
240  std::smatch matches;
241  if (std::regex_match(line, matches, pattern))
242  return (int64_t)(atof(matches[2].str().c_str()) * 1000000);
243  }
244 
245  return -1;
246 #elif defined(_WIN32) || defined(_WIN64)
247  HKEY hKeyProcessor;
248  long lError = RegOpenKeyExA(HKEY_LOCAL_MACHINE, "HARDWARE\\DESCRIPTION\\System\\CentralProcessor\\0", 0, KEY_READ, &hKeyProcessor);
249  if (lError != ERROR_SUCCESS)
250  return -1;
251 
252  // Smart resource cleaner pattern
253  auto key = resource(hKeyProcessor, [](HKEY hKey) { RegCloseKey(hKey); });
254 
255  DWORD dwMHz = 0;
256  DWORD dwBufferSize = sizeof(DWORD);
257  lError = RegQueryValueExA(key.get(), "~MHz", nullptr, nullptr, (LPBYTE)&dwMHz, &dwBufferSize);
258  if (lError != ERROR_SUCCESS)
259  return -1;
260 
261  return dwMHz * 1000000;
262 #else
263 #error Unsupported platform
264 #endif
265  }
