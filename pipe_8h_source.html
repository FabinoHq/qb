105  inline auto *allocate_back(std::size_t const size) {
106  if (likely(_end + size < _capacity)) {
107  const auto save_index = _end;
108  _end += size;
109  return _data + save_index;
110  }
111  const auto nb_item = _end - _begin;
112  const auto half = _capacity / 2;
113  if (_begin > half && size < half) {
114  reorder();
115  _end += size;
116  return _data + nb_item;
117  } else {
118  std::size_t new_capacity;
119  do {
120  _factor <<= 1;
121  new_capacity = _factor * _SIZE;
122  } while (new_capacity - nb_item < size);
123 
124  const auto new_data = base_type::allocate(new_capacity);
125  std::memcpy(new_data, _data + _begin, nb_item * sizeof(T));
126  base_type::deallocate(_data, _capacity);
127 
128  _begin = 0;
129  _end = nb_item + size;
130  _capacity = new_capacity;
131  _data = new_data;
132  }
133  return _data + nb_item;
134  }
